    function sendMessage() {
      let input = document.getElementById("input");
      let chat = document.getElementById("chat");
      let userText = input.value.trim();
      if (!userText) return;

      chat.innerHTML += "<p><b>You:</b> " + userText + "</p>";
      input.value = "";

      // Simple bot replies
      let botReply = "I donâ€™t understand.";
      if (userText.toLowerCase().includes("hello")) botReply = "Hi there!";
      if (userText.toLowerCase().includes("how are you")) botReply = "I'm just code, but I'm good!";
      if (userText.toLowerCase().includes("bye")) botReply = "Goodbye!";

      chat.innerHTML += "<p><b>Bot:</b> " + botReply + "</p>";
      chat.scrollTop = chat.scrollHeight;
    }
  </script>
